262       function upload(data, type) {
263         if (!options.contentType) options.contentType = type || defaultType;
264         APICall.binaryUpload(apicall, data, options.filename, options.contentType).done();
265       }

267       if (isString(file) || (Buffer && file instanceof Buffer)) {
268         // Upload by filename
269

271           if (isString(file)) file = require('fs').readFileSync(file);

278       } else if (CanvasRenderingContext2D && file instanceof CanvasRenderingContext2D) {

280       } else if (isBinary(file)) {

283

288 
289         /** @private */

293 
294         /** @private */
295         reader.onload = function(e) {
296           upload(e.target.result);
297         }
298 
299         // Don't need to transform Files to Blobs.
300         if (File && file instanceof File) {
301           if (!options.contentType && file.type != "") options.contentType = file.type;
302         } else {
303           file = getBlob(file, options.contentType || defaultType);
304         }
305         
306         reader.readAsDataURL(file);
307       } else NotSupported();
308

312     /**
313      * Download a file stored in CloudMine.
314      * @param {string} key The binary file's object key.
315      * @param {object} options Override defaults set on WebService. See WebService constructor for parameters.
316      * @config {string} [filename] If present, the file will be downloaded directly to the computer with the
317      *                             filename given. This does not validate the filename given!
318      * @config {string} [mode] If buffer, automatically move returning data to either an ArrayBuffer or Buffer
319      *                         if supported. Otherwise the result will be a standard string.
320      * @return {APICall} An APICall instance for the web service request used to attach events.
321      */
322     download: function(key, options) {
323       options = opts(this, options);
324       var response = {success: {}}, query;
325 
326       if (options.filename) {
327         query = {
328           force_download: true,
329           apikey: options.apikey,
330           session_token: options.session_token,
331           filename: options.filename
332         }
333       }
334 
335       var apicall = new APICall({
336         action: 'binary/' + key,
337         type: 'GET',
338         later: true,
339         encoding: 'binary',
340         options: options,
341         query: query,
342         appid: options.appid,
343         apikey: options.apikey
344       });
345 
346       // Download file directly to computer if given a filename.
347       if (options.filename) {
348         if (ajax == NodeAJAX) {
349           apicall.setProcessor(function(data) {
350             response.success[key] = require('fs').writeFileSync(options.filename, data, 'binary');
351             return response;
352           }).done();
353         } else {
354           function detach() {
355             if (iframe.parentNode) document.body.removeChild(iframe);
356           }
357           var iframe = document.createElement('iframe');
358           iframe.style.display = 'none';
359           iframe.src = apicall.url;
360           iframe.onload = function() { setTimeout(detach, 5000); };
361           detach.timer = setTimeout(detach, 60000);
362           document.body.appendChild(iframe);
363           response.success[key] = iframe;
364         }
365 
366         apicall.done(response);
367       } else if (options.mode === 'buffer' && (ArrayBuffer || Buffer)) {
368         apicall.setProcessor(function(data) {
369           var buffer;
370           if (Buffer) {
371             buffer = new Buffer(data, 'binary');
372           } else {
373             buffer = new ArrayBuffer(data.length);
374             var charView = new Uint8Array(buffer);
375             for (var i = 0; i < data.length; ++i) {
376               charView[i] = data[i] & 0xFF;
377             }
378           }
379 
380           response.success[key] = buffer;
381           return response;
382         }).done();
383       } else {
384         // Raw data return. Do not attempt to process the result.
385         apicall.setProcessor(function(data) {
386           response.success[key] = data;
387           return response;
388         }).done();
389       }
390 
391       return apicall;
392     },

723    * what were attached.

729    *       requires a small subset of jQuery-like AJAX functionality.

731    * <p>Event firing order:
732    *    <div>Successes: HTTP Code String, HTTP Code Number, 'success'</div>
733    *    <div>Meta: 'meta' - This is for operations that can write meta data.</div>
734    *    <div>Result: 'result' - This is results from code snippets if used.</div>
735    *    <div>Errors: HTTP Code String, HTTP Code Number, 'error'</div>

737    * <p>Event callback signatures:
738    *    <div>HTTP Codes: function(keys, responseObject, statusCode)</div>
739    *    <div>'error': function(keys, responesObject)</div>
740    *    <div>'success': function(keys, responseObject)</div>
741    *    <div>'meta': function(keys, responseObject)</div>
742    *    <div>'result: function(keys, responseObject)</div>
743    *    <div>'abort': function(keys, responseObject)</div>
744    * @class
745    * @name APICall

772     this.status = null;

776

778     var root = '/', session = opts.session_token, applevel = opts.applevel;
779     if (applevel === false || (applevel !== true && session != null)) {
780       root = '/user/';

782     }

789     /** @private */
790     this.config.complete = function(xhr, status) {
791       var data;
792       if (xhr) {
793         data = xhr.responseText
794         self.status = xhr.status;
795         self.responseText = data;
796         each(xhr.getAllResponseHeaders().split('\n'), function(item) {
797           var index = (item.indexOf(':'));
798           if (index > 0) self.responseHeaders[item.substring(0, index)] = item.substring(index + 2);
799         });
800 
801         // If we can parse the data as JSON or store the original data.
802         try {
803           self.data = JSON.parse(data || "{}");
804         } catch (e) {
805           self.data = data;
806         }
807       } else {
808         self.status = 'abort';
809         self.data = [ sConfig.data ];
810       }
811 
812       // Parse the response only if a safe status
813       if (status == 'success' && self.status >= 200 && self.status < 300) {
814         // Preprocess data coming in to hash-hash: [success/errors].[httpcode]
815         data = sConfig.processResponse.call(self, self.data, xhr, self);
816       } else {
817         data = {errors: {}};
818         if (isString(self.data)) {
819           self.data = { errors: [ self.data ] } // This way, we can rely on this structure data[statuscode].errors to always be an Array of one or more errors
820         }
821         data.errors[self.status] = self.data;
822       }
823 
824       setTimeout(function() {
825         APICall.complete(self, data);
826       }, 1);
827     }
828 
829     // Let script continue before triggering ajax call
830     if (!this.config.later && this.config.async) {
831       setTimeout(function() {
832         self.xhr = ajax(self.url, sConfig);
833       }, 1);
834     }
835   }

984   /**
985    * Complete the given API Call, usually called after completed processing of data, though can be
986    * used to circumvent the standard AJAX call functionality for calls that are not currently running.
987    * @param {APICall} apicall The api call to affect. Can be either a completed request or a deferred request.
988    * @param {object} data Processed data where the top level keys are: success, errors, meta, result.
989    * 
990    * @private
991    * @function
992    * @memberOf APICall
993   */
994   APICall.complete = function(apicall, data) {
995     // Success results may have errors for certain keys
996     if (data.errors) apicall.hasErrors = true;
997 
998     // Clean up temporary state.
999     if (apicall.config) {
1000       apicall.config = undefined;
1001       delete apicall.config;
1002     }
1003     if (apicall.xhr) {
1004       apicall.xhr = undefined;
1005       delete apicall.xhr;
1006     }
1007 
1008     // Data has been processed by this point and should exist in success, errors, meta, or results hashes.
1009     // Event firing order: http status (e.g. ok, created), http status (e.g. 200, 201), success, meta, result, error.
1010     if (data.success) {
1011       // Callback signature: function(keys, response, statusCode)
1012       if (httpcode[apicall.status]) apicall.trigger(httpcode[apicall.status], data.success, apicall, apicall.status);
1013       apicall.trigger(apicall.status, data.success, apicall, apicall.status);
1014 
1015       // Callback signature: function(keys, response);
1016       apicall.trigger('success', data.success, apicall);
1017     }
1018 
1019     // Callback signature: function(keys, response)
1020     if (data.meta) apicall.trigger('meta', data.meta, apicall);
1021 
1022     // Callback signature: function(keys, response)
1023     if (data.result) apicall.trigger('result', data.result, apicall);
1024 
1025     // Errors needs to fire groups of errors depending on code result.
1026     if (data.errors) {
1027       // Callback signature: function(keys, reponse, statusCode)
1028       for (var k in data.errors) {
1029         if (httpcode[k]) apicall.trigger(httpcode[k], data.errors[k], apicall, k);
1030         apicall.trigger(k, data.errors[k], apicall, k);
1031       }
1032 
1033       // Callback signature: function(keys, response)
1034       apicall.trigger('error', data.errors, apicall);
1035     }
1036 
1037     // Callback signature: function(responseData, response)
1038     apicall.trigger('complete', data, apicall);      
1039   }

1042   /**
1043    * Standard CloudMine response for the 200-299 range responses.
1044    * This will transform the response so that APICall.complete will trigger the appropriate handlers.
1045    * 
1046    * @private
1047    * @function
1048    * @memberOf APICall
1049   */
1050   APICall.textResponse = function(data, xhr, response) {
1051     var out = {};
1052     if (data.success || data.errors || data.meta || data.result) {
1053       if (data.count != null) response.count = data.count;
1054       if (!isEmptyObject(data.success)) out.success = data.success;
1055       if (!isEmptyObject(data.meta)) out.meta = data.meta;
1056       if (!isEmptyObject(data.result)) out.result = data.result;
1057       if (!isEmptyObject(data.errors)) {
1058         out.errors = {};
1059         for (var k in data.errors) {
1060           var error = data.errors[k], code = 400;
1061 
1062           // Unfortunately, errors are a bit inconsistent.
1063           if (error.code) {
1064             code = error.code;
1065             error = error.message || error;
1066           } else if (isString(error)) {
1067             code = errors[error.toLowerCase()] || 400;
1068           }
1069 
1070           if (!out.errors[code]) out.errors[code] = {}
1071           out.errors[code][k] = {errors: [ error ]};
1072         }
1073       }
1074 
1075       // At least guarantee a success callback.
1076       if (isEmptyObject(out)) {
1077         if (this.config.options && this.config.options.snippet) out.result = {};
1078         else out.success = {};
1079       }
1080     } else {
1081       // Non-standard response. Just pass back the data we were given.
1082       out = {success: data};
1083     }
1084 
1085     return out;
1086   }

1139   /**
1140    * Node.JS jQuery-like ajax adapter.
1141    * This is an internal class that is not exposed.
1142    *
1143    * @param {string} uri The complete url to hit.
1144    * @param {object} config Parameters for the ajax request.
1145    * @config {string} [contentType] The Content Type of the request.
1146    * @config {string} [type] The type of request, e.g. 'get', 'post'
1147    * @config {object} [headers] Request headers to send to the client.
1148    * @config {boolean} [processData] If true, process the data given.
1149    * @config {string|Array|Object} [data] Data to send to the server.
1150    * @name HttpRequest
1151    * @constructor
1152    */
1153   function HttpRequest(uri, config) {
1154     config = config || {};
1155     this.status = 400;
1156     this.responseText = [];
1157     this._headers = {};
1158 
1159     // disable connection pooling
1160     // disable chunked transfer-encoding which nginx doesn't support
1161     var opts = url.parse(uri);
1162     opts.agent = false;
1163     opts.method = config.type;
1164     opts.headers = config.headers || {};
1165 
1166     // Preprocess data if it is JSON data.
1167     if (isObject(config.data) && config.processData) {
1168       config.data = JSON.stringify(config.data);
1169     }
1170 
1171     // Attach a content-length
1172     if (isArray(config.data)) opts.headers['content-length'] = Buffer.byteLength(config.data);
1173     else if (isString(config.data)) opts.headers['content-length'] = config.data.length;
1174     
1175     // Fire request.
1176     var self = this, cbContext = config.context || this;
1177     this._textStatus = 'success';
1178     this._request = (opts.protocol === "http:" ? http : https).request(opts, function(response) {
1179       response.setEncoding(config.encoding || 'utf8');
1180 
1181       response.on('data', function(chunk) {
1182         self.responseText.push(chunk);
1183       });
1184 
1185       response.on('close', function() {
1186         response.emit('end');
1187       });
1188 
1189       response.on('end', function() {
1190         self._headers = stringify(response.headers);
1191         self.status = response.statusCode;
1192 
1193         // Process data if necessary.
1194         var data = self.responseText = self.responseText.join('');
1195         if (config.dataType == 'json' || (config.dataType != 'text' && response.headers['content-type'].match(/\bapplication\/json\b/i))) {
1196           try {
1197             data = JSON.parse(data);
1198           } catch (e) {
1199             self._textStatus = 'parsererror';
1200           }
1201         }
1202 
1203         
1204         if (self._textStatus == 'success' && self.status >= 200 && self.status < 300) {
1205           if (config.success) config.success.call(cbContext, data, 'success', self);
1206         } else if (config.error) {
1207           config.error.call(cbContext, self, 'error', self.responseText);
1208         }
1209         if (config.complete) config.complete.call(cbContext, self, self._textStatus);
1210       });
1211     });
1212 
1213     // Handle request errors.
1214     this._request.on('error', function(e) {
1215       self.status = e.status;
1216       self.responseText = e.message;
1217       self._textStatus = 'error';
1218       if (config.error) config.error.call(cbContext, self, 'error', e.message);
1219       if (config.complete) config.complete.call(cbContext, self, 'error');
1220     });
1221 
1222     // Send data (if present) and fire the request.
1223     this._request.end(config.data);
1224   }
